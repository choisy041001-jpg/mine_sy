토스트

화면 하단 중앙에 노출되는 알림입니다. JS 함수 showToastV7(msg) 를 호출하여 사용합니다. 인자는 메시지 문자열 1개만 받습니다.

.toast-v7 기본 저장·삭제 완료 알림
저장이 완료되었습니다.
HTML — body 직전에 한 번만 배치
<div id="toastV7" class="toast-v7"></div>
JavaScript
// 함수 정의
function showToastV7(msg) {
  const t = document.getElementById('toastV7');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2800);
}

// 호출 — 인자 1개만, sub 인자 없음
showToastV7('저장이 완료되었습니다.');

  dashboard:       { title:'대시보드',     crumb:'관리자 / 대시보드' },
  leave:           { title:'연차 관리',     crumb:'관리자 / 연차 관리' },
  cert:            { title:'업무신청관리', crumb:'관리자 / 업무신청관리' },
  faq:             { title:'FAQ 관리',     crumb:'관리자 / FAQ 관리' },
  'emp-dashboard': { title:'대시보드',     crumb:'직원 / 대시보드' },
  'emp-cert':      { title:'업무신청',     crumb:'직원 / 업무신청' },
  'emp-profile':   { title:'프로필 관리', crumb:'직원 / 프로필 관리' },
};

// nav-item onclick="goPage(this)"
function goPage(navItem) {
  const page = navItem.dataset.page;
  document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
  navItem.classList.add('active');
  document.querySelectorAll('.page').forEach(el => el.classList.remove('active'));
  document.querySelector('.page[data-page="' + page + '"]').classList.add('active');
  if (PAGE_META[page]) {
    pageTitle.textContent = PAGE_META[page].title;
    pageCrumb.textContent = PAGE_META[page].crumb;
  }
}

// .sbp-option onclick="switchRole('admin')" 또는 switchRole('emp')
function switchRole(role) {
  const p = SIDEBAR_PROFILES[role];
  sbpAvatar.textContent = p.avatar;
  sbpName.textContent   = p.name;
  sbpRole.textContent   = p.role;

  const isAdmin = (role === 'admin');
  navAdmin.style.display   = isAdmin ? '' : 'none';
  navDivider.style.display = isAdmin ? '' : 'none';
  navBottom.style.display  = isAdmin ? '' : 'none';

  goPageByName(isAdmin ? 'dashboard' : 'emp-dashboard');
}
